יִשְׂרָאֵ֑ל אֶל ־הָעִ֨יר הַזֹּ֜את אֲשֶׁ֣ר ׀ אַתֶּ֣ם אֹמְרִ֗ים נִתְּנָה֙ בְּיַ֣ד מֶֽלֶךְ ־בָּבֶ֔ל בַּחֶ֖רֶב וּבָרָעָ֥ב וּבַדָּֽבֶר ׃ 37 הִנְנִ֤י מְקַבְּצָם֙ מִכָּל ־הָ֣אֲרָצ֔וֹת אֲשֶׁ֨ר הִדַּחְתִּ֥ים שָׁ֛ם בְּאַפִּ֥י וּבַחֲמָתִ֖י וּבְקֶ֣צֶף גָּד֑וֹל וַהֲשִֽׁבֹתִים֙ אֶל ־הַמָּק֣וֹם הַזֶּ֔ה וְהֹשַׁבְתִּ֖ים לָבֶֽטַח ׃ 38 וְהָ֥יוּ לִ֖י לְעָ֑ם וַאֲנִ֕י אֶהְיֶ֥ה לָהֶ֖ם לֵאלֹהִֽים ׃ 39 וְנָתַתִּ֨י לָהֶ֜ם לֵ֤ב אֶחָד֙ וְדֶ֣רֶךְ אֶחָ֔ד לְיִרְאָ֥ה אוֹתִ֖י כָּל ־הַיָּמִ֑ים לְט֣וֹב לָהֶ֔ם וְלִבְנֵיהֶ֖ם אַחֲרֵיהֶֽם ׃ 40 וְכָרַתִּ֤י לָהֶם֙ בְּרִ֣ית עוֹלָ֔ם אֲשֶׁ֤ר לֹֽא ־אָשׁוּב֙ מֵאַ֣חֲרֵיהֶ֔ם לְהֵיטִיבִ֖י אוֹתָ֑ם וְאֶת ־יִרְאָתִי֙ אֶתֵּ֣ן בִּלְבָבָ֔ם לְבִלְתִּ֖י ס֥וּר מֵעָלָֽי ׃ 41 וְשַׂשְׂתִּ֥י עֲלֵיהֶ֖ם לְהֵטִ֣יב אוֹתָ֑ם וּנְטַעְתִּ֞ים בָּאָ֤רֶץ הַזֹּאת֙ בֶּאֱמֶ֔ת בְּכָל ־לִבִּ֖י וּבְכָל ־נַפְשִֽׁי ׃ס 42 כִּֽי ־כֹה֙ אָמַ֣ר יְהוָ֔ה כַּאֲשֶׁ֤ר הֵבֵ֙אתִי֙ אֶל ־הָעָ֣ם הַזֶּ֔ה אֵ֛ת כָּל ־הָרָעָ֥ה הַגְּדוֹלָ֖ה הַזֹּ֑את כֵּ֣ן אָנֹכִ֞י מֵבִ֤יא עֲלֵיהֶם֙ אֶת ־כָּל ־הַטּוֹבָ֔ה אֲשֶׁ֥ר אָנֹכִ֖י דֹּבֵ֥ר עֲלֵיהֶֽם ׃ 43 וְנִקְנָ֥ה הַשָּׂדֶ֖ה בָּאָ֣רֶץ הַזֹּ֑את אֲשֶׁ֣ר ׀ אַתֶּ֣ם אֹמְרִ֗ים שְׁמָמָ֥ה הִיא֙ מֵאֵ֤ין אָדָם֙ וּבְהֵמָ֔ה נִתְּנָ֖ה בְּיַ֥ד הַכַּשְׂדִּֽים ׃ 44 שָׂד֞וֹת בַּכֶּ֣סֶף יִקְנ֗וּ וְכָת֨וֹב בַּסֵּ֥פֶר ׀ וְחָתוֹם֮ וְהָעֵ֣ד עֵדִים֒ בְּאֶ֨רֶץ בִּנְיָמִ֜ן וּבִסְבִיבֵ֣י יְרוּשָׁלִַ֗ם וּבְעָרֵ֤י יְהוּדָה֙ וּבְעָרֵ֣י הָהָ֔ר וּבְעָרֵ֥י הַשְּׁפֵלָ֖ה וּבְעָרֵ֣י הַנֶּ֑גֶב כִּֽי ־אָשִׁ֥יב
